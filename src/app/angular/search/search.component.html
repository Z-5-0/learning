<input [value]="searchText" (input)="onSearch($event)"

      // [value] - Property binding - Az input mező megkapja a class property-jének default értékét.
      // (input) - Event binding - Ha történik event, egy event object emittálódik, ami a $event változóban érhető el.
      // A fenti két binding ötvözése two-way data binding-ot eredményez.
    
      <input [(ngModel)]="searchText">
      // Beépített direktíva.
    
      <input [ngModel]="searchText" (ngModelChange)="searchText = $event"> // = [(ngModel)]="searchText"
      // Ebben az esetben szétszedtük a kimenetet és a bemenetet, de mivel ugyanazt a változót (searchText) használjuk, így megmarad a two-way data binding!

Search result for: {{ searchText }}
